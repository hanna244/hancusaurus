## 문제 
아홉 난쟁이의 키가 주어졌을 때, 백설공주를 도와 일곱 난쟁이를 찾는 프로그램을 작성하시오.

## 로직

```js
function solution(arr) {
  let answer = arr
  let sum = answer.reduce((a, b) => a + b, 0)
  for (let i = 0; i < 8; i++) {
    for (let j = i + 1; j < 9; j++) {
      if (sum - (answer[i] + answer[j]) == 100) {
        answer.splice(j, 1)
        answer.splice(i, 1)
      }
    }
  }
  return answer
}

solution([20, 7, 23, 19, 10, 15, 25, 8, 13])
```
